STEP 7.5 — POLLING GATE: wait for backend PR to merge into develop
    echo "Waiting for backend PR (branch feature/be-{{FEATURE_SLUG}}) to merge..."
    WAIT_START=$(date +%s)
    MAX_WAIT=3600    # 60 minutes hard cap — if hit, STOP and ping Arvin.

    while true; do
        # Primary gate: backend PR state from GitHub's API.
        BE_STATE=$(gh pr list \
            --head "feature/be-{{FEATURE_SLUG}}" \
            --base develop \
            --state all \
            --json state,mergeCommit \
            --jq '.[0].state // "NONE"')

        case "$BE_STATE" in
            MERGED)
                echo "Backend PR merged. Syncing develop and proceeding."
                break
                ;;
            OPEN)
                # Surface backend CI status so we see trouble early.
                BE_CI=$(gh pr list --head "feature/be-{{FEATURE_SLUG}}" \
                    --json statusCheckRollup --jq \
                    '.[0].statusCheckRollup | map(.conclusion) | unique | join(",")' \
                    2>/dev/null || echo "unknown")
                echo "$(date +%T) Backend PR open, CI: ${BE_CI}. Waiting 60s..."
                ;;
            CLOSED)
                echo "Backend PR closed without merging. STOP — ask Arvin."
                exit 1
                ;;
            NONE)
                echo "$(date +%T) Backend PR not yet opened. Waiting 60s..."
                ;;
        esac

        # Timeout guard
        NOW=$(date +%s)
        if [ $((NOW - WAIT_START)) -gt $MAX_WAIT ]; then
            echo "Waited > ${MAX_WAIT}s. STOP — investigate the backend session."
            exit 1
        fi

        sleep 60
    done

    # Fast-forward local develop with the merged backend commit so our
    # rebase (if needed) is against the right base.
    git fetch --all --prune
    git fetch origin develop:develop

    # If origin/develop moved past our branch base, rebase our feature branch
    # on top. This keeps the squash merge clean.
    if ! git merge-base --is-ancestor develop feature/fe-{{FEATURE_SLUG}}; then
        echo "Rebasing feature/fe-{{FEATURE_SLUG}} onto fresh develop..."
        git rebase develop
        git push --force-with-lease origin feature/fe-{{FEATURE_SLUG}}
    fi

STEP 7.6 — Mark the PR ready and re-trigger CI
    gh pr ready "$FE_PR"
    echo "Marked PR #$FE_PR as ready for review."

STEP 8 — Wait for CI green, then merge
    gh pr checks --watch "$FE_PR"
    gh pr merge "$FE_PR" --squash --delete-branch --admin
    # If admin merge refused, STOP and tell Arvin.

STEP 9 — Resync worktree to fresh develop
    git fetch --all --prune
    git fetch origin develop:develop || {
        echo "Could not fast-forward local develop after merge. STOP."; exit 1;
    }
    if git switch develop 2>/dev/null; then
        echo "On local develop."
    else
        git switch --detach develop
        echo "develop checked out in other worktree; this one is detached."
    fi
    git branch -D feature/fe-{{FEATURE_SLUG}} 2>/dev/null || true
    git status
    git log --oneline -5 develop

STEP 10 — Comment on backend PR and report back
    MERGE_SHA=$(git rev-parse develop)

    gh pr comment "$BE_PR" --body "Frontend merged at \`$MERGE_SHA\`. Full feature ({{FEATURE_SLUG}}) is now on develop — safe to promote to main." \
        2>/dev/null || true
